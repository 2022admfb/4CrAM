ra_fild12,295,208,0	script	toairplane#rachel	45,1,1,{
	end;
OnTouch:
	mes "ในการใช้เรือเหาะคุณจะต้องจ่าย 1,200 zeny หรือตั๋วเรือเหาะฟรี";
	mes "คุณต้องการใช้บริการหรือไม่";
	next;
	if (select("ตกลง:ยกเลิก") == 1) {
		if (countitem(7311) > 0) {
			delitem 7311,1; //Free_Flying_Ship_Ticket
			warp "airplane_01",245,60;
			end;
		}
		if (Zeny >= 1200) {
			set Zeny, Zeny - 1200;
			warp "airplane_01",245,60;
			end;
		}
		mes "ฉันขอโทษ แต่คุณมีเงินไม่พอ";
		mes "โปรดจำไว้ว่าคุณต้องจ่าย 1,200 zeny เพื่อใช้บริการ";
		close;
	}
	mes "ขอบคุณครับ ไว้มาใหม่";
	close;
}
