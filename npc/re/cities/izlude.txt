function	script	F_IzludeChannel	{
	mes "[Channel Warp Official]";
	mes "Izlude เป็นสถานที่ที่มีชีวิตชีวาจริงๆใช่หรือไม่";
	mes "ฉันมาที่นี่เพื่อช่วยให้ Izlude ราบรื่นขึ้น!";
	mes "ฉันสามารถส่ง Izlude อื่นให้คุณได้";
	mes "คุณต้องการไปที่ Channel อื่นหรือไม่";
	next;
	set .@i, select(getarg(0)+":สิ้นสุดการสนทนา");
	if (.@i < 6) {
		setarray .@maps$[1],"izlude","izlude_a","izlude_b","izlude_c","izlude_d";
		savepoint .@maps$[.@i],128,98,1,1;
		warp .@maps$[.@i],128,98;
		end;
	}
	close;
}
