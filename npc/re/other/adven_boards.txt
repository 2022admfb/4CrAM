function	script	F_AvenBoard	{
	.@mo_board = select( "เลือกสถานที่",
					"ค้นหา มอนเตอร์",
					( countitem(909) == 30000 ? "คำขอ":"" ),
					"ยกเลิก" );
	if (.@mo_board == 1)
		return 1;
	else if (.@mo_board == 2) {
		while(1) {
			mes "[Monster Information Retrieval System]";
			mes "นี่คือระบบดึงข้อมูลมอนสเตอร์ ข้อมูลของมอนสเตอร์ที่ปรากฏใน ดันเจี้ยนนี้สามารถค้นหาได้ ป้อนชื่อ มอนสเตอร์รวมถึงช่องว่างเมื่อค้นหา";
			next;
			input .@inputstr$;
			setarray .@element$[0],"Neutral","Water","Earth","Fire","Wind","Poison","Holy","Shadow","Ghost","Undead";
			setarray .@size$[0],"Medium","Small","Big";
			setarray .@race$[0],"Formless","Undead","Animal","Plant","Insect","Fish","Demon","Demihuman","Angel","Dragon";
			.@size = getargcount();
			.@i = 0;
			while( .@i < .@size && getarg(.@i+1,"") != .@inputstr$ ) .@i += 3;
			mes "[Monster Information Retrieval System]";
			mes "กำลังค้นหา";
			next;
			mes "[Monster Information Retrieval System]";
			mes "การค้นหาเสร็จสมบูรณ์แล้ว";
			mes "กำลังแสดงผลการค้นหา";
			next;
			mes "[Monster Information Retrieval System]";
			if (.@i >= .@size) 
			{
				mes "ไม่มีข้อมูลสำหรับมอนสเตอร์ตัวนั้น";
				mes "โปรดตรวจสอบชื่อมอนสเตอร์ของคุณ";
			}
			else 
			{
				.@mob_id = getarg(.@i);
				.@lv$ = getmonsterinfo(.@mob_id,MOB_LV);
				mes "- ไอดี:"+ .@mob_id;
				mes "- ชื่อ:"+ getarg(.@i+1);
				mes "- เลเวล:"+ .@lv$;
				mes "- ธาตุ:"+ .@element$[ getmonsterinfo(.@mob_id,MOB_ELEMENT) ];
				mes "- ขนาด:"+ .@size$[ getmonsterinfo(.@mob_id,MOB_SIZE) ];
				mes "- ประเภท:"+ .@race$[ getmonsterinfo(.@mob_id,MOB_RACE) ];
				//mes "- ไอเท็ม :"+ .@item[.@count]; 
				mes "-------------------------------- ^FD8D02";
				mes getarg(.@i+2);
			}
			next;
				if(getmobdrops(.@mob_id))
				{
					.@count = $@MobDrop_count;
					copyarray	.@item[0],$@MobDrop_item[0],.@count;
					copyarray	.@rate[0],$@MobDrop_rate[0],.@count;
					mes "[ รายละเอียด Item drop] ";
					for( .@i = 0; .@i < .@count; .@i++ ) 
					{
					mes .@item[.@i] + " (" + getitemname(.@item[.@i]) + ") " + .@rate[.@i]/100 + ((.@rate[.@i]%100 < 10) ? ".0":".") + .@rate[.@i]%100 + "%";
					}
					next;
				}
			mes "[ ระบบ ค้นหา ข้อมูล มอนเตอร์]";
			mes "คุณต้องการค้นหาข้อมูลสัตว์ประหลาดอื่น ๆ หรือไม่?";
			next;
			if (select( "ค้นหา","ยกเลิก" ) == 1)
				continue;
			mes "[Monster Information Retrieval System]";
			mes "ระบบค้นหามอนสเตอร์จะปิดตัวลง";
			next;
			return 2;
			}	
		}
	else if (.@mo_board == 3) {
		mes "ข้อมูลที่ร้องขอจะปรากฏขึ้น - ไม่มีปัญหาหากปรากฏ";
		next;
		return 3;
	}
	else if (.@mo_board == 4)
		close;
	else {
		mes "ข้อผิดพลาดในการทำงาน";
		close;
	}
}

	next;
	return;
}
