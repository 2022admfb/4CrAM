-	script	Kriemhild#1::OutsideFlagsPR1	722,{

	set .@GID, GetCastleData("prtg_cas01",1);

	if (.@GID == 0) {
		mes "[ Edict of the Divine Rune-Midgarts Kingdom ]";
		mes " ";
		mes "1. ตามพิธีการของ The Divine Rune-Midgarts Kingdom, ";
		mes "เราขอประกาศว่า";
		mes "ไม่มีผู้ครอบครอง สำหรับ ปราสาท นี้.";
		mes " ";
		mes "2. แด่คนที่ทำได้ ";
		mes "เอาชนะ";
		mes "และทำลาย Emperium,";
		mes "กษัตริย์จะประทาน";
		mes "เป็นเจ้าของปราสาทนี้.";
		close;
	}
	if (getcharid(2) == .@GID) {
		mes "[ Echoing Voice ]";
		mes "ผู้กล้า ...";
		mes "คุณต้องการกลับไปยังสถานที่ที่มีเกียรติของคุณหรือไม่";
		next;
		if(select("กลับไปที่ปราสาทกิลด์.:ยกเลิก.") == 1) {
			close2;
			if (getcharid(2) == GetCastleData("prtg_cas01",1)) {
				warp "prtg_cas01",96,173;
				end;
			}
			end;
		}
		close;
	}
	mes "[ Edict of the Divine Rune-Midgarts Kingdom ]";
	mes " ";
	mes "1. ตามพิธีการของ The Divine Rune-Midgarts Kingdom,, ";
	mes "พื้นที่ แห่งนี้";
	mes "เป็น ปราสาท ของ ^ff0000"+GetGuildName(.@GID)+"^000000 Guild.";
	mes " ";
	mes "2. หัวหน้า กิลด์ ของ ^ff0000"+GetGuildName(.@GID)+"^000000 Guild คือ";
	mes "^ff0000"+GetGuildMaster(.@GID)+"^000000";
	mes "ถ้ามีใครคัดค้าน,";
	mes "พิสูจน์ความแข็งแกร่งและเกียรติยศของคุณด้วยดาบเหล็กในมือ";
	close;

OnRecvCastlePt01:
	FlagEmblem GetCastleData("prtg_cas01",1);
	end;
}
